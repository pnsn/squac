MetricComparator
Default value : ( propA: RowMetric, propB: RowMetric ): number => { if (propA.value === null && propB.value === null) { // both are null, treat as equal return 0; } else if (propA.value === null && propB.value !== null) { // treat second value as larger return -1; } else if (propA.value !== null && propB.value === null) { // treat first as larger return 1; } return propA.value - propB.value; }
